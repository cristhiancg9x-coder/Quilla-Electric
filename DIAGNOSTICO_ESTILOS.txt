import MainLayout from '@/layouts/MainLayout.astro';
import CctvWizard from '@/modules/calculator/components/CctvWizard.jsx';

// 1. Importamos la base de datos cruda
import productsData from '@modules/calculator/data/cctv-products.json';

// 2. Importamos la herramienta de optimización de Astro
import { getImage } from "astro:assets";

// 3. Cargamos TODAS las imágenes de la carpeta assets (Mapeo automático)
const imageFiles = import.meta.glob('/src/assets/camaras/*.{jpeg,jpg,png,tiff,webp,gif,svg}');

// 4. PROCESO DE AUTOMATIZACIÓN (Esto ocurre en milisegundos al construir)
const optimizedProducts = await Promise.all(
  productsData.map(async (product) => {
    // Construimos la ruta donde debería estar la imagen
    const imagePath = `/src/assets/camaras/${product.image}`;
    
    // Verificamos si la imagen existe en la carpeta
    if (!imageFiles[imagePath]) {
      console.error(`⚠️ Alerta: No encontré la imagen ${product.image} en src/assets/camaras`);
      return product; // Si no existe, devolvemos el producto sin cambios (evita errores)
    }

    // Si existe, la procesamos
    const imageModule = await imageFiles[imagePath]() as any;
    const optimizedImage = await getImage({
      src: imageModule.default,
      format: 'webp', // ¡Conversión automática!
      width: 600,     // Redimensionado automático
      quality: 80     // Compresión inteligente
    });

    // Devolvemos el producto con la NUEVA ruta optimizada
    return {
      ...product,
      image: optimizedImage.src 
    };
  })
);
